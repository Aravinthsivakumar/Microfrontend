import { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import api from '../api/axios';

const CATEGORIES = ['Web Development','Data Science','Cloud Computing',
                    'Mobile Development','DevOps','Cybersecurity'];

export default function UpgradeForm() {
  const [form, setForm] = useState({ skills: '', category: '', experience: '' });
  const [cert, setCert] = useState(null);
  const [status, setStatus] = useState('idle'); // idle | loading | success | error
  const [errMsg, setErrMsg] = useState('');
  const navigate = useNavigate();

  const set = (k, v) => setForm(f => ({ ...f, [k]: v }));

  const handleSubmit = async (e) => {
    e.preventDefault();
    setStatus('loading');
    const fd = new FormData();
    fd.append('data', new Blob([JSON.stringify({
      skills: form.skills,
      category: form.category,
      experience: parseInt(form.experience) || 0,
    })], { type: 'application/json' }));
    if (cert) fd.append('certificate', cert);
    try {
      await api.post('/user/upgrade-request', fd, {
        headers: { 'Content-Type': 'multipart/form-data' }
      });
      setStatus('success');
      setTimeout(() => navigate('/profile'), 2500);
    } catch (e) {
      setErrMsg(e.response?.data?.error || 'Submission failed');
      setStatus('error');
    }
  };

  return (
    <div>
      <div className="page-header">
        <h5 className="mb-0 fw-bold">Upgrade to Trainer</h5>
      </div>

      <div className="px-4 pb-4">
        <div className="row justify-content-center">
          <div className="col-md-7">
            {status === 'success' ? (
              <div className="card text-center p-5">
                <div className="mx-auto mb-3 rounded-circle d-flex align-items-center justify-content-center"
                     style={{ width: 80, height: 80, background: '#d1fae5' }}>
                  <i className="bi bi-check-circle-fill text-success" style={{ fontSize: '2.5rem' }}></i>
                </div>
                <h5 className="fw-bold mb-2">Request Submitted!</h5>
                <p className="text-muted mb-3">
                  Your trainer upgrade request is under review. You'll be notified once an admin
                  processes it. Redirecting to profile…
                </p>
                <div className="spinner-border spinner-border-sm text-primary mx-auto"></div>
              </div>
            ) : (
              <>
                {/* Info banner */}
                <div className="card mb-4 border-0"
                     style={{ background: 'linear-gradient(135deg,#4f46e5,#7c3aed)',
                              borderRadius: 14 }}>
                  <div className="card-body text-white p-4">
                    <div className="d-flex gap-3">
                      <i className="bi bi-award-fill fs-2 opacity-75"></i>
                      <div>
                        <h6 className="fw-bold mb-1">Become a CourseConnect Trainer</h6>
                        <p className="mb-0 opacity-80 small">
                          Share your expertise, create courses, and earn money. Fill in your
                          details and our admin team will review within 24–48 hours.
                        </p>
                      </div>
                    </div>
                  </div>
                </div>

                <div className="card">
                  <div className="card-body p-4">
                    {status === 'error' && (
                      <div className="alert alert-danger d-flex align-items-center gap-2 py-2">
                        <i className="bi bi-exclamation-circle-fill"></i> {errMsg}
                      </div>
                    )}

                    <form onSubmit={handleSubmit}>
                      <div className="mb-3">
                        <label className="form-label fw-semibold small">
                          Skills <span className="text-danger">*</span>
                        </label>
                        <input type="text" className="form-control"
                          placeholder="e.g. React, Node.js, Python, Machine Learning"
                          value={form.skills}
                          onChange={e => set('skills', e.target.value)}
                          required />
                        <div className="form-text">Separate multiple skills with commas</div>
                      </div>

                      <div className="mb-3">
                        <label className="form-label fw-semibold small">
                          Category <span className="text-danger">*</span>
                        </label>
                        <select className="form-select" value={form.category}
                          onChange={e => set('category', e.target.value)} required>
                          <option value="">Select a category…</option>
                          {CATEGORIES.map(c => (
                            <option key={c} value={c}>{c}</option>
                          ))}
                        </select>
                      </div>

                      <div className="mb-3">
                        <label className="form-label fw-semibold small">
                          Years of Experience <span className="text-danger">*</span>
                        </label>
                        <input type="number" className="form-control"
                          placeholder="e.g. 5"
                          min="0" max="50"
                          value={form.experience}
                          onChange={e => set('experience', e.target.value)}
                          required />
                      </div>

                      <div className="mb-4">
                        <label className="form-label fw-semibold small">
                          Certificate / Credential
                          <span className="badge bg-secondary ms-2" style={{ fontSize: '0.65rem' }}>
                            Optional
                          </span>
                        </label>
                        <div className="border border-dashed rounded-3 p-3 text-center"
                             style={{ borderColor: '#cbd5e1', background: '#f8fafc' }}>
                          {cert ? (
                            <div className="d-flex align-items-center justify-content-center gap-2">
                              <i className="bi bi-file-earmark-check text-success fs-4"></i>
                              <div>
                                <p className="mb-0 small fw-semibold">{cert.name}</p>
                                <small className="text-muted">
                                  {(cert.size / 1024).toFixed(1)} KB
                                </small>
                              </div>
                              <button type="button" className="btn btn-sm btn-outline-danger ms-2"
                                onClick={() => setCert(null)}>
                                <i className="bi bi-x"></i>
                              </button>
                            </div>
                          ) : (
                            <>
                              <i className="bi bi-cloud-upload text-muted fs-3 mb-2 d-block"></i>
                              <p className="text-muted small mb-2">
                                Drag & drop or click to upload
                              </p>
                              <label className="btn btn-sm btn-outline-primary">
                                Choose File
                                <input type="file" accept=".pdf,.jpg,.jpeg,.png"
                                  className="d-none"
                                  onChange={e => setCert(e.target.files[0])} />
                              </label>
                              <p className="text-muted mt-2 mb-0" style={{ fontSize: '0.75rem' }}>
                                PDF, JPG, PNG up to 5MB
                              </p>
                            </>
                          )}
                        </div>
                      </div>

                      {/* Checklist */}
                      <div className="card bg-light border-0 p-3 mb-4">
                        <p className="small fw-semibold mb-2">Before submitting, confirm:</p>
                        {[
                          'I have at least 1 year of relevant experience',
                          'I will update my phone number after approval',
                          'I understand that admin approval is required',
                        ].map((item, i) => (
                          <div key={i} className="form-check">
                            <input className="form-check-input" type="checkbox"
                              id={`check${i}`} required />
                            <label className="form-check-label small text-muted" htmlFor={`check${i}`}>
                              {item}
                            </label>
                          </div>
                        ))}
                      </div>

                      <button type="submit" className="btn btn-primary w-100 py-2"
                              disabled={status === 'loading'}>
                        {status === 'loading'
                          ? <><span className="spinner-border spinner-border-sm me-2"></span>
                              Submitting…</>
                          : <><i className="bi bi-send-fill me-2"></i>Submit Request</>}
                      </button>
                    </form>
                  </div>
                </div>
              </>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
